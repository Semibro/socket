        • ver1 : bias = True
        • ver2 : bias = False

ver1 : loss = sum of squares error 

alpha = 1 / 1000
What = torch.tensor([[-5.0], [10.0]], requires_grad = True)

for epoch in range(30):
    # step 1 : yhat
    yhat = X @ What
    # step 2 : loss
    loss = torch.sum((y - yhat) ** 2)
    # step 3 : 미분
    loss.backward()
    # step 4 : update
    What.data = What.data - alpha * What.grad
    What.grad = None
